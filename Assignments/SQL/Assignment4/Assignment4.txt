-- 1. T-SQL Program to find factorial of a given number
CREATE DATABASE Assignment4;


USE Assignment4;


DECLARE @num INT = 5;
DECLARE @fact BIGINT = 1;
DECLARE @i INT = 1;

WHILE @i <= @num
BEGIN
    SET @fact = @fact * @i;
    SET @i = @i + 1;
END;

PRINT 'Factorial of ' + CAST(@num AS VARCHAR) + ' is ' + CAST(@fact AS VARCHAR);



-- 2. Stored Procedure for multiplication table
GO
CREATE PROCEDURE sp_multiplicationtable
    @num INT,
    @limit INT
AS
BEGIN
    DECLARE @i INT = 1;

    WHILE @i <= @limit
    BEGIN
        PRINT CAST(@num AS VARCHAR(10)) + ' x ' +
              CAST(@i AS VARCHAR(10)) + ' = ' +
              CAST(@num * @i AS VARCHAR(10));

        SET @i = @i + 1;
    END
END;


-- Execute procedure
EXEC sp_multiplicationtable 5, 10;

-- Creating function
GO
CREATE FUNCTION fn_studentstatus (@score INT)
RETURNS VARCHAR(10)
AS
BEGIN
    DECLARE @status VARCHAR(10);

    IF @score >= 50
        SET @status = 'PASS';
    ELSE
        SET @status = 'FAIL';

    RETURN @status;
END;

-- Display student data with status
SELECT 
    s.sid,
    s.sname,
    m.score,
    dbo.fn_studentstatus(m.score) AS status
FROM student s
JOIN marks m ON s.sid = m.sid;
